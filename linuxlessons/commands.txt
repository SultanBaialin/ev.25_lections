linux= commands 
ctrl + alt + T - команда открытия терминала
 ~ - симвом домашней директории
 / - символ корневой директории 
 .. - обозначает предыдущие директории 
 . -  обозначает текущую директорию 
 && - сразу выполняет две или несколько команд
 	
    1. ls(list) - команда для вывода всех директорий(папка) и файлов в текущей директории в терминале 
    2. cd(change desktop) - команда для изменения(переходов) одной дериктории на другую 
    			cd <название дериктории>
    3. mkdir(make directory) - создает дерикторию в рабочей директории в терминале			
    4. touch - создает указаный файл 
    5. rm - удаление файлов 
    6. rmdir - удаление директории
    7. pwr (present working directory) - выводит путь до текущей директории
    8. rm -rf - принудительное удаление 
    9. ls -a - выводит список как скрытых так и обычных файлов и директорий
    10. nano <path_to_file> - откроется файл для редактирования в графическом интерфейсе
  *если нет файла, то nano попытается создать новый файл
    11. cat <fil_name> - выводит содержимое файла в терминал 
    12. echo - вводит текст внутри файла 
    		echo текст > название файла 
    13. mv - перемещение файлов
    переименовывание: mv john john 
           	      mv <старое_название> <новое_название> 
    перемещение: mv john ~/Desktop/
                 mv <file/directory> <directory>/
    14. ls -lh - выводит обьем занимаемой памяти 
    15. sudo - утилита для выполнения команд суперпользователя
    16. exit - закрытие терминала
    17. sudo useradd - добавление нового пользователя
